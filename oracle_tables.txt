-- ðŸ”¹ SEQUENCES
DROP SEQUENCE PATIENT_SEQ;
CREATE SEQUENCE PATIENT_SEQ START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;

DROP SEQUENCE DOCTOR_SCHEDULE_SEQ;
CREATE SEQUENCE DOCTOR_SCHEDULE_SEQ START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;

DROP SEQUENCE EMERGENCY_CASES_SEQ;
CREATE SEQUENCE EMERGENCY_CASES_SEQ START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;

-- ðŸ”¹ USERS1 TABLE

CREATE TABLE USERS1 (
    USER_ID VARCHAR2(50) PRIMARY KEY,
    NAME VARCHAR2(100),
    ROLE VARCHAR2(50) NOT NULL CHECK (ROLE IN ('Doctor', 'Patient', 'Admin')),
    PASSWORD VARCHAR2(255) NOT NULL,
    SPECIALIZATION VARCHAR2(50)
);

-- ðŸ”¹ DOCTORS TABLE

CREATE TABLE DOCTORS (
    DOCTOR_ID VARCHAR2(20) PRIMARY KEY,
    STATUS VARCHAR2(20) DEFAULT 'Available'
);

-- ðŸ”¹ DOCTOR_SCHEDULE TABLE

CREATE TABLE DOCTOR_SCHEDULE (
    DOCTOR_ID VARCHAR2(20),
    DOCTOR_NAME VARCHAR2(100),
    SPECIALIZATION VARCHAR2(50),
    AVAILABLE_DAYS VARCHAR2(50),
    FOREIGN KEY (DOCTOR_ID) REFERENCES USERS1(USER_ID)
);

-- ðŸ”¹ PATIENT_TRIAGE TABLE

CREATE TABLE PATIENT_TRIAGE (
    PATIENT_ID VARCHAR2(10) PRIMARY KEY,
    NAME VARCHAR2(100) NOT NULL,
    AGE NUMBER NOT NULL,
    GENDER VARCHAR2(10) NOT NULL,
    CONTACT NUMBER(15) NOT NULL,
    SYMPTOMS CLOB NOT NULL,
    MEDICAL_HISTORY CLOB,
    TRIAGE_LEVEL VARCHAR2(50),
    ESTIMATED_WAITING_TIME VARCHAR2(50),
    BP VARCHAR2(20),
    HEART_RATE NUMBER,
    RESPIRATORY_RATE NUMBER,
    OXYGEN_SATURATION NUMBER,
    SPECIALIZATION VARCHAR2(50),
    APPOINTMENT_TIME VARCHAR2(20),
    STATUS VARCHAR2(20) DEFAULT 'Pending',
    ASSIGNED_DOCTOR VARCHAR2(100),
    BOOKING_DATETIME TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    APPOINTMENT_DATE DATE NOT NULL
);

-- ðŸ”¹ TRIGGER FOR PATIENT_ID
CREATE OR REPLACE TRIGGER PATIENT_ID_TRIGGER
BEFORE INSERT ON PATIENT_TRIAGE
FOR EACH ROW
BEGIN
    SELECT 'P' || LPAD(PATIENT_SEQ.NEXTVAL, 3, '0') INTO :NEW.PATIENT_ID FROM DUAL;
END;
/

-- ðŸ”¹ PPD TABLE (Prescriptions)

CREATE TABLE PPD (
    PATIENT_ID VARCHAR2(10),
    PATIENT_NAME VARCHAR2(100),
    DOCTOR_ID VARCHAR2(10),
    DOCTOR_NAME VARCHAR2(100),
    SPECIALIZATION VARCHAR2(50),
    CRITICALITY VARCHAR2(20),
    APPOINTMENT_DATE DATE,
    APPOINTMENT_TIME VARCHAR2(10),
    PRESCRIBED_MEDICINES VARCHAR2(500),
    PRESCRIPTION_DETAILS VARCHAR2(1000),
    ADDITIONAL_NOTES VARCHAR2(1000)
);

-- ðŸ”¹ EMERGENCY_QUEUE TABLE

CREATE TABLE EMERGENCY_QUEUE (
    EMERGENCY_ID NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    PATIENT_NAME VARCHAR2(100),
    PHONE VARCHAR2(15),
    SPECIALIZATION VARCHAR2(100),
    ASSIGNED_DOCTOR VARCHAR2(20),
    STATUS VARCHAR2(20) DEFAULT 'Waiting',
    REPORTED_AT TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_emergency_doctor FOREIGN KEY (ASSIGNED_DOCTOR) REFERENCES USERS1(USER_ID)
);

-- ðŸ”¹ WALKINS TABLE

CREATE TABLE WALKINS (
    TOKEN_NO NUMBER PRIMARY KEY,
    PATIENT_NAME VARCHAR2(100),
    AGE NUMBER,
    GENDER VARCHAR2(10),
    CONTACT VARCHAR2(20),
    SYMPTOMS CLOB,
    SPECIALIZATION VARCHAR2(50),
    TRIAGE_LEVEL VARCHAR2(20),
    ASSIGNED_DOCTOR VARCHAR2(50),
    APPOINTMENT_DATE DATE,
    APPOINTMENT_TIME VARCHAR2(10),
    ESTIMATED_WAITING_TIME VARCHAR2(20),
    STATUS VARCHAR2(20),
    BOOKING_DATETIME TIMESTAMP
);
